   148→  btnConnect.addEventListener('click', async () => {
   149→    const raw = cookieInput.value.trim();
   150→    if (!raw) return;
   151→
   152→    connectError.style.display = 'none';
   153→    connectSpinner.style.display = 'inline-block';
   154→    connectText.textContent = 'Connecting…';
   155→    btnConnect.disabled = true;
   156→
   157→    try {
   158→      if (!isWails || !window.go) {
   159→        toast('Running in browser mode — Wails bindings unavailable. Run as desktop app.', 'error');
   160→        connectText.textContent = 'Connect';
   161→        connectSpinner.style.display = 'none';
   162→        btnConnect.disabled = false;
   163→        return;
   164→      }
   165→
   166→      const result = await window.go.main.App.Authenticate(raw);
   167→
   168→      if (result.error) {
   169→        connectError.textContent = result.error;
   170→        connectError.style.display = 'block';
   171→        toast(result.error, 'error');
   172→        return;
   173→      }
   174→
   175→      // Store state
   176→      state.cookieJSON = raw;
   177→      state.authenticated = true;
   178→      state.userInfo = result.userInfo;
   179→      state.vipInfo = result.vipInfo;
   180→      state.restPoint = result.restPoint;
   181→
   182→      // Load models
   183→      try {
   184→        const modelsResult = await window.go.main.App.GetModels();
   185→        if (!modelsResult.error) {
   186→          state.models = modelsResult.models || [];
   187→          state.scenes = modelsResult.scenes || [];
   188→          console.log(`[Models] Loaded ${state.models.length} models, ${state.scenes.length} scenes`);
   189→        } else {
   190→          console.error('[Models] Error:', modelsResult.error);
   191→        }
   192→      } catch (e) {
   193→        console.error('Failed to load models:', e);
   194→      }
   195→
   196→      showAuthenticatedView();
   197→      toast('Connected successfully!', 'success');
   198→    } catch (err) {
   199→      connectError.textContent = err.message || 'Connection failed';
   200→      connectError.style.display = 'block';
   201→      toast(err.message || 'Connection failed', 'error');
   202→    } finally {
   203→      connectText.textContent = 'Connect';
   204→      connectSpinner.style.display = 'none';
   205→      btnConnect.disabled = false;
   206→    }

   292→  function updateModelUI() {
   293→    const model = state.models.find((m) => m.modelName === state.modelName);
   294→    if (!model) return;
   295→
   296→    // Duration
   297→    groupDuration.innerHTML = '';
   298→    const durations = model.duration || [{ value: 5 }];
   299→    durations.forEach((d) => {
   300→      const btn = document.createElement('button');
   301→      btn.textContent = d.value + 's';
   302→      if (d.value === state.duration || (!durations.find((x) => x.value === state.duration))) {
   303→        btn.classList.add('active');
   304→        state.duration = d.value;
   305→      }
   306→      btn.addEventListener('click', () => {
   307→        groupDuration.querySelectorAll('button').forEach((b) => b.classList.remove('active'));
   308→        btn.classList.add('active');
   309→        state.duration = d.value;
   310→        selectAiType(model);
   311→      });
   312→      groupDuration.appendChild(btn);
   313→    });
   314→
   315→    // Resolution
   316→    groupResolution.innerHTML = '';
   317→    const resolutions = model.videoResolution || ['720', '1080'];
   318→    resolutions.forEach((r) => {
   319→      const btn = document.createElement('button');
   320→      btn.textContent = r + 'p';
   321→      if (r === state.resolution) btn.classList.add('active');
   322→      btn.addEventListener('click', () => {
   323→        groupResolution.querySelectorAll('button').forEach((b) => b.classList.remove('active'));
   324→        btn.classList.add('active');
   325→        state.resolution = r;
   326→        selectAiType(model);
   327→      });
   328→      groupResolution.appendChild(btn);
   329→    });
   330→
   331→    // Video Size
   332→    groupVideoSize.innerHTML = '';
   333→    const sizes = model.videoSize || [{ ratio: '16:9' }, { ratio: '9:16' }, { ratio: '1:1' }];
   334→    sizes.forEach((s) => {
   335→      const btn = document.createElement('button');
   336→      btn.textContent = s.ratio;
   337→      if (s.ratio === state.videoSize) btn.classList.add('active');
   338→      btn.addEventListener('click', () => {
   339→        groupVideoSize.querySelectorAll('button').forEach((b) => b.classList.remove('active'));
   340→        btn.classList.add('active');
   341→        state.videoSize = s.ratio;
   342→      });
   343→      groupVideoSize.appendChild(btn);
   344→    });
   345→
   346→    // Motion Duration
   347→    if (state.sceneId === 'motion' && model.pointCostMotion) {
   348→      groupMotDuration.innerHTML = '';
   349→      const motDurations = [...new Set(model.pointCostMotion.map((c) => c.motDuration))];
   350→      motDurations.forEach((md) => {
   351→        const btn = document.createElement('button');
   352→        btn.textContent = md + 's';
   353→        if (String(md) === state.motDuration) btn.classList.add('active');
   354→        btn.addEventListener('click', () => {
   355→          groupMotDuration.querySelectorAll('button').forEach((b) => b.classList.remove('active'));
   356→          btn.classList.add('active');
   357→          state.motDuration = String(md);
   358→          selectAiType(model);
   359→        });
   360→        groupMotDuration.appendChild(btn);
   361→      });
   362→    }
   363→
   364→    // Auto-select AI type
   365→    selectAiType(model);
   366→  }
   367→
   368→  function selectAiType(model) {
   369→    state.aiType = 0;
   370→    if (state.sceneId === 'motion') {
   371→      const match = (model.pointCostMotion || []).find(
   372→        (c) => String(c.motDuration) === state.motDuration && c.resolution === state.resolution
   373→      );
   374→      if (match) state.aiType = match.aiType;
   375→    } else if (state.sceneId === 'reference') {
   376→      const match = (model.pointCostReference || []).find(
   377→        (c) => c.duration === state.duration && c.resolution === state.resolution
   378→      );
   379→      if (match) state.aiType = match.aiType;
   380→    } else {
   381→      const match = (model.pointCostImage || []).find(
   382→        (c) => c.duration === state.duration && c.resolution === state.resolution
   383→      );
   384→      if (match) state.aiType = match.aiType;
   385→    }
   386→    console.log(`[Config] aiType=${state.aiType} (scene=${state.sceneId}, duration=${state.duration}, res=${state.resolution})`);
   387→  }

   405→          state.imageFilePath = path;
   406→          // Read file and preview via Go backend (returns data URL)
   407→          window.go.main.App.ReadFileAsDataURL(path).then((dataURL) => {
   408→            if (dataURL) {
   409→              previewImage.src = dataURL; // Already a complete data:image/...;base64,... URL
   410→              previewImage.style.display = 'block';
   411→              btnClearImage.style.display = 'flex';
   412→              zoneImage.querySelector('.upload-placeholder').style.display = 'none';
   413→            } else {
   414→              console.error('[Upload] Failed to read image file for preview');
   415→              toast('Failed to preview image', 'error');
   416→            }
   417→          }).catch((err) => {
   418→            console.error('[Upload] ReadFileAsDataURL error:', err);
   419→          });
   420→        }
   421→      }).catch((err) => {
   422→        console.error('[Upload] PickImageFile error:', err);
   423→      });
   424→    } else {
   425→      // Browser fallback: trigger hidden file input
   426→      fileImage.click();
   427→    }
   428→  }

   599→  async function startGeneration() {
   600→    if (state.isGenerating) return;
   601→
   602→    if (!isWails || !window.go) {
   603→      toast('Wails bindings required — run as desktop app', 'error');
   604→      return;
   605→    }
   606→
   607→    // Validate inputs
   608→    if (!state.modelName) {
   609→      toast('Please select a model', 'error');
   610→      return;
   611→    }
   612→
   613→    const needsImage = state.sceneId === 'text_or_image' || state.sceneId === 'motion';
   614→    const needsVideo = state.sceneId === 'motion';
   615→    const needsAnyFile = state.sceneId === 'reference';
   616→
   617→    if (needsImage && !state.imageFilePath) {
   618→      toast('Please upload a source image', 'error');
   619→      return;
   620→    }
   621→
   622→    if (needsVideo && !state.videoFilePath) {
   623→      toast('Please upload a motion video', 'error');
   624→      return;
   625→    }
   626→
   627→    if (needsAnyFile && !state.imageFilePath && !state.videoFilePath) {
   628→      toast('Please upload at least one image or video', 'error');
   629→      return;
   630→    }
   631→
   632→    state.isGenerating = true;
   633→    btnGenerate.disabled = true;
   634→    generateText.textContent = 'Generating…';
   635→    generateSpinner.style.display = 'inline-block';
   636→    progressSection.style.display = 'block';
   637→    resultSection.style.display = 'none';
   638→    progressFill.style.width = '0%';
   639→    progressFill.style.background = '';
   640→    progressText.textContent = 'Uploading files…';
   641→
   642→    try {
   643→      const model = state.models.find((m) => m.modelName === state.modelName);
   644→      if (model) selectAiType(model);
   645→
   646→      console.log('[Generate] Starting generation with:', {
   647→        sceneId: state.sceneId,
   648→        modelName: state.modelName,
   649→        duration: state.duration,
   650→        resolution: state.resolution,
   651→        videoSize: state.videoSize,
   652→        aiType: state.aiType,
   653→        image: state.imageFilePath || '(none)',
   654→        video: state.videoFilePath || '(none)',
   655→        prompt: inputPrompt.value || '(empty)',
   656→      });
   657→
   658→      const result = await window.go.main.App.GenerateVideo(
   659→        state.imageFilePath || '',
   660→        state.videoFilePath || '',
   661→        inputPrompt.value,
   662→        state.sceneId,
   663→        state.modelName,
   664→        state.duration,
   665→        state.resolution,
   666→        state.videoSize,
   667→        state.aiType,
   668→        state.motDuration,
   669→        state.keepOriginalSound
   670→      );
   671→
   672→      console.log('[Generate] Result:', result);
   673→
   674→      if (result.error) {
   675→        toast('Generation failed: ' + result.error, 'error');
   676→        progressText.textContent = 'Failed: ' + result.error;
   677→        progressFill.style.width = '100%';
   678→        progressFill.style.background = 'var(--danger)';
   679→        return;
   680→      }
   681→
   682→      if (result.docId) {
   683→        state.currentDocId = result.docId;
   684→        progressText.textContent = 'Task submitted — processing…';
   685→        progressFill.style.width = '20%';
   686→        startPolling();
   687→        toast('Task submitted! Waiting for result...', 'success');
   688→      } else if (result.chatId) {
   689→        // SSE stream mode: use chatId for history-based polling
   690→        state.currentDocId = result.chatId;
   691→        progressText.textContent = 'Task submitted — processing…';
   692→        progressFill.style.width = '20%';
   693→        startPolling();
   694→        toast('Task submitted! Waiting for result...', 'success');
   695→      } else {
   696→        // No docId/chatId but no error either — check submitResult for clues
   697→        console.warn('[Generate] No docId/chatId returned. submitResult:', result.submitResult);
   698→        toast('Task submitted but no ID received. Check Go console for details.', 'info');
   699→        progressText.textContent = 'Submitted (no ID — check console)';
   700→      }
   701→    } catch (err) {
   702→      console.error('[Generate] Exception:', err);
   703→      toast('Error: ' + (err.message || 'Unknown error'), 'error');
   704→      progressText.textContent = 'Error: ' + (err.message || 'Unknown');
   705→      progressFill.style.width = '100%';
   706→      progressFill.style.background = 'var(--danger)';
   707→    } finally {
   708→      state.isGenerating = false;
   709→      btnGenerate.disabled = false;
   710→      generateText.textContent = 'Generate Video';
   711→      generateSpinner.style.display = 'none';
   712→    }
   713→  }

   718→  function startPolling() {
   719→    stopPolling();
   720→    state.pollInterval = setInterval(pollTaskStatus, 3000);
   721→    // Poll immediately
   722→    pollTaskStatus();
   723→  }
   724→
   725→  function stopPolling() {
   726→    if (state.pollInterval) {
   727→      clearInterval(state.pollInterval);
   728→      state.pollInterval = null;
   729→    }
   730→  }
   731→
   732→  async function pollTaskStatus() {
   733→    if (!state.currentDocId || !window.go) return;
   734→
   735→    try {
   736→      const result = await window.go.main.App.GetTaskStatus(state.currentDocId);
   737→
   738→      if (result.error) {
   739→        // If polling returns error (e.g. params error from server), 
   740→        // try refreshing history instead of giving up immediately
   741→        console.warn('[Poll] Status error:', result.error);
   742→        const pollCount = parseInt(progressText.dataset.pollCount || '0') + 1;
   743→        progressText.dataset.pollCount = pollCount;
   744→        
   745→        if (pollCount >= 5) {
   746→          stopPolling();
   747→          progressText.textContent = 'Task submitted. Check History tab for results.';
   748→          progressFill.style.width = '100%';
   749→          progressFill.style.background = 'var(--warning, #f59e0b)';
   750→          toast('Video may be processing. Check the History tab.', 'info');
   751→          loadHistory();
   752→        }
   753→        return;
   754→      }
   755→
   756→      const statusData = result.data || {};
   757→      const status = statusData.status;
   758→
   759→      console.log(`[Poll] docId=${state.currentDocId} status=${status} progress=${statusData.progress}`);
   760→
   761→      if (status === 1) {
   762→        // Processing
   763→        const progress = statusData.progress || 50;
   764→        progressFill.style.width = Math.min(90, 20 + progress * 0.7) + '%';
   765→        progressText.textContent = 'Processing… ' + (progress ? progress + '%' : '');
   766→      } else if (status === 2) {
   767→        // Completed
   768→        stopPolling();
   769→        progressFill.style.width = '100%';
   770→        progressText.textContent = 'Completed!';
   771→
   772→        const videoUrl = statusData.videoUrl || statusData.video_url || statusData.url;
   773→        if (videoUrl) {
   774→          resultVideo.src = videoUrl;
   775→          resultSection.style.display = 'block';
   776→          toast('Video generated successfully!', 'success');
   777→        } else {
   778→          toast('Completed but no video URL in response', 'info');
   779→          console.log('[Poll] Completed status data:', statusData);
   780→        }
   781→
   782→        // Update credits
   783→        try {
   784→          const authResult = await window.go.main.App.Authenticate(state.cookieJSON);
   785→          if (!authResult.error) {
   786→            state.restPoint = authResult.restPoint;
   787→            statCredits.textContent = authResult.restPoint;
   788→            headerCreditsValue.textContent = authResult.restPoint;
   789→          }
   790→        } catch (e) { /* ignore */ }
   791→
   792→        // Refresh history
   793→        loadHistory();
   794→      } else if (status === 3 || status === -1 || status === 4) {
   795→        // Failed
   796→        stopPolling();
   797→        progressFill.style.width = '100%';
   798→        progressFill.style.background = 'var(--danger)';
   799→        const errMsg = statusData.errorMsg || statusData.error || statusData.errMsg || 'Unknown error';
   800→        progressText.textContent = 'Failed: ' + errMsg;
   801→        toast('Generation failed: ' + errMsg, 'error');
   802→        console.error('[Poll] Failed status data:', statusData);
   803→        setTimeout(() => { progressFill.style.background = ''; }, 3000);
   804→      }
   805→    } catch (err) {
   806→      console.error('[Poll] Error:', err);
   807→    }
   808→  }

   813→  async function loadHistory() {
   814→    if (!window.go) return;
   815→
   816→    btnRefreshHistory.disabled = true;
   817→    historyEmpty.style.display = 'none';
   818→
   819→    // Remove existing items (not the empty state)
   820→    historyList.querySelectorAll('.history-item').forEach((el) => el.remove());
   821→
   822→    try {
   823→      const result = await window.go.main.App.GetHistory(1, 50);
   824→
   825→      if (result.error) {
   826→        historyEmpty.style.display = 'block';
   827→        return;
   828→      }
   829→
   830→      state.history = result.items || [];
   831→
   832→      if (state.history.length === 0) {
   833→        historyEmpty.style.display = 'block';
   834→        statHistory.textContent = '0';
   835→        statCompleted.textContent = '0';
   836→        return;
   837→      }
   838→
   839→      statHistory.textContent = state.history.length;
   840→      statCompleted.textContent = state.history.filter((h) => h.status === 2).length;
   841→
   842→      state.history.forEach((item) => {
   843→        const el = document.createElement('div');
   844→        el.className = 'history-item';
   845→
   846→        const statusClass = item.status === 2 ? 'completed' : item.status === 1 ? 'processing' : 'failed';
   847→        const statusText = item.status === 2 ? 'Done' : item.status === 1 ? 'Processing' : 'Failed';
   848→        const date = new Date(item.createTime * 1000).toLocaleDateString();
   849→
   850→        el.innerHTML = `
   851→          <div class="history-thumb">
   852→            ${item.thumbnailUrl ? '<img src="' + item.thumbnailUrl + '" />' : '<svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5"><polygon points="5 3 19 12 5 21 5 3"/></svg>'}
   853→          </div>
   854→          <div class="history-info">
   855→            <p class="history-title">${escapeHTML(item.title || item.prompt || 'Untitled')}</p>
   856→            <div class="history-meta">
   857→              <span>${item.modelName || '-'}</span>
   858→              <span>${date}</span>
   859→              <span class="badge-status ${statusClass}">${statusText}</span>
   860→            </div>
   861→          </div>
   862→        `;
   863→
   864→        if (item.videoUrl) {
   865→          el.addEventListener('click', () => openVideoDialog(item.videoUrl, item.title || 'Video'));
   866→        }
   867→
   868→        historyList.appendChild(el);
   869→      });
   870→    } catch (e) {
   871→      console.error('[History] Error:', e);
   872→      historyEmpty.style.display = 'block';
   873→    } finally {
   874→      btnRefreshHistory.disabled = false;
   875→    }
   876→  }
