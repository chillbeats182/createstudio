   725‚Üí  async function pollTaskStatus() {
   726‚Üí    if (!state.currentDocId || !window.go) return;
   727‚Üí
   728‚Üí    try {
   729‚Üí      const result = await window.go.main.App.GetTaskStatus(state.currentDocId);
   730‚Üí
   731‚Üí      if (result.error) {
   732‚Üí        // If polling returns error (e.g. params error from server), 
   733‚Üí        // try refreshing history instead of giving up immediately
   734‚Üí        console.warn('[Poll] Status error:', result.error);
   735‚Üí        const pollCount = parseInt(progressText.dataset.pollCount || '0') + 1;
   736‚Üí        progressText.dataset.pollCount = pollCount;
   737‚Üí        
   738‚Üí        if (pollCount >= 5) {
   739‚Üí          stopPolling();
   740‚Üí          progressText.textContent = 'Task submitted. Check History tab for results.';
   741‚Üí          progressFill.style.width = '100%';
   742‚Üí          progressFill.style.background = 'var(--warning, #f59e0b)';
   743‚Üí          toast('Video may be processing. Check the History tab.', 'info');
   744‚Üí          loadHistory();
   745‚Üí        }
   746‚Üí        return;
   747‚Üí      }
   748‚Üí
   749‚Üí      const statusData = result.data || {};
   750‚Üí      const status = statusData.status;
   751‚Üí
   752‚Üí      console.log(`[Poll] docId=${state.currentDocId} status=${status} progress=${statusData.progress}`);
   753‚Üí
   754‚Üí      if (status === 1) {
   755‚Üí        // Processing
   756‚Üí        const progress = statusData.progress || 50;
   757‚Üí        progressFill.style.width = Math.min(90, 20 + progress * 0.7) + '%';
   758‚Üí        progressText.textContent = 'Processing‚Ä¶ ' + (progress ? progress + '%' : '');
   759‚Üí      } else if (status === 2) {
   760‚Üí        // Completed
   761‚Üí        stopPolling();
   762‚Üí        progressFill.style.width = '100%';
   763‚Üí        progressText.textContent = 'Completed!';
   764‚Üí
   765‚Üí        const videoUrl = statusData.videoUrl || statusData.video_url || statusData.url;
   766‚Üí        if (videoUrl) {
   767‚Üí          resultVideo.src = videoUrl;
   768‚Üí          resultSection.style.display = 'block';
   769‚Üí          toast('Video generated successfully!', 'success');
   770‚Üí        } else {
   771‚Üí          toast('Completed but no video URL in response', 'info');
   772‚Üí          console.log('[Poll] Completed status data:', statusData);
   773‚Üí        }
   774‚Üí
   775‚Üí        // Update credits
   776‚Üí        try {
   777‚Üí          const authResult = await window.go.main.App.Authenticate(state.cookieJSON);
   778‚Üí          if (!authResult.error) {
   779‚Üí            state.restPoint = authResult.restPoint;
   780‚Üí            statCredits.textContent = authResult.restPoint;
   781‚Üí            headerCreditsValue.textContent = authResult.restPoint;
   782‚Üí          }
   783‚Üí        } catch (e) { /* ignore */ }
   784‚Üí
   785‚Üí        // Refresh history
   786‚Üí        loadHistory();
   787‚Üí      } else if (status === 3 || status === -1 || status === 4) {
   788‚Üí        // Failed
   789‚Üí        stopPolling();
   790‚Üí        progressFill.style.width = '100%';
   791‚Üí        progressFill.style.background = 'var(--danger)';
   792‚Üí        const errMsg = statusData.errorMsg || statusData.error || statusData.errMsg || 'Unknown error';
   793‚Üí        progressText.textContent = 'Failed: ' + errMsg;
   794‚Üí        toast('Generation failed: ' + errMsg, 'error');
   795‚Üí        console.error('[Poll] Failed status data:', statusData);
   796‚Üí        setTimeout(() => { progressFill.style.background = ''; }, 3000);
   797‚Üí      }
   798‚Üí    } catch (err) {
   799‚Üí      console.error('[Poll] Error:', err);
   800‚Üí    }
   801‚Üí  }
